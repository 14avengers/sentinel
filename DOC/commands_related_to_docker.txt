BUILDING DOCKER IMAGE:
    docker build --tag sentinel --compress .

RUNNING DOCKER CONTAINER:
    docker run -it -p 30303:30303 -p 30303:30303/udp sentinel

REMOVING ALL IMAGES AND CONTAINERS:
    docker rmi $(docker images -a -q ) -f; \
    docker stop $(docker ps -a -q); \
    docker rm $(docker ps -a -q); \
    sudo rm -rf /var/lib/docker/aufs/diff

KNOW RUNNING CONTAINERS:
    docker ps -a

RESUMING PREVIOUS CONTAINER:
    docker start -i CONTAINER_ID
